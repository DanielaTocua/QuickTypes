New Entity Person {
    properties {
        number id :{ primary, generated},
        string name ,
        string lastname
    }
}

New Entity Animal {
    properties {
        number id :{ primary, generated},
        string name,
        string breed,
        number age ,
        number owner_id
    }
    relations {
        ManyToOne Person (owner_id, id): {
            onDelete: cascade

        }

    }
}
